	on_startup = {
		## Reichspakt
		effect = {
			GER = {
				if = {
					limit = {
						BEL = {
							is_ai = yes
						}
					}
					annex_country = {
						target = BEL
						transfer_troops = yes
					}
				}
				every_owned_state = {
					limit = {
						OR = {
							is_core_of = POL
							is_core_of = BEL
							is_core_of = LIT
						}
					}
					add_resistance_target = {
						amount = -50
						occupier = GER
					}
				}
			}
			UKR = {
				annex_country = {
					target = WHR
					transfer_troops = yes
				}
				every_owned_state = {
					limit = {
						is_core_of = WHR
						has_resistance = yes
					}
					add_compliance = 50
				}
			}
		}
		effect = {
			FNG = {
				add_ideas = {
					low_economic_mobilisation
				}
			}
			JAP = {
				add_political_power = 50
			}
		}
		## Global
		effect = {
			every_country = {
				limit = {
					is_ai = no
					has_navy_size = {
						size > 0
						type = capital_ship
					}
				}
				country_event = {
					id = jck.navy.0
				}
			}
		}
	}
